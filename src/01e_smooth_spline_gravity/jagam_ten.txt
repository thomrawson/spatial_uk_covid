model {
  eta <- X %*% b + offset ## linear predictor
  for (i in 1:n) { mu[i] <-  exp(eta[i]) } ## expected response
  for (i in 1:n) { y[i] ~ dpois(mu[i]) } ## response 
  ## Parametric effect priors CHECK tau=1/60^2 is appropriate!
  for (i in 1:1) { b[i] ~ dnorm(0,0.00028) }
  ## prior for s(y_scale,x_scale)... 
  K1 <- S1[1:39,1:39] * lambda[1]  + S1[1:39,40:78] * lambda[2]
  b[2:40] ~ dmnorm(zero[2:40],K1) 
  ## prior for s(connect_coord1_scale,connect_coord2_scale)... 
  K2 <- S2[1:39,1:39] * lambda[3]  + S2[1:39,40:78] * lambda[4]
  b[41:79] ~ dmnorm(zero[41:79],K2) 
  ## smoothing parameter priors CHECK...
  for (i in 1:4) {
    lambda[i] ~ dgamma(.05,.005)
    rho[i] <- log(lambda[i])
  }
}
